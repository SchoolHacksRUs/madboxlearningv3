function makeParticle(type) {
  const W = bgCanvas.width, H = bgCanvas.height;
  switch(type) {
    case 'snow': return {
      x: Math.random()*W, y: Math.random()*H,
      r: 0.4 + Math.random()*Math.random()*2.6,
      op: 0.04 + Math.random()*0.14, sp: 0.12 + Math.random()*0.4,
      dr: (Math.random()-.5)*0.4, wb: Math.random()*Math.PI*2, color: 'rgba(255,255,255,'
    };
    case 'ash': return {
      x: Math.random()*W, y: Math.random()*H,
      r: 0.6 + Math.random()*2, op: 0.03 + Math.random()*0.1,
      sp: 0.06 + Math.random()*0.2, dr: (Math.random()-.5)*0.6,
      wb: Math.random()*Math.PI*2, rot: Math.random()*Math.PI*2,
      rotSpeed: (Math.random()-.5)*0.03, isRect: Math.random() > 0.4,
      color: `rgba(${150+Math.floor(Math.random()*50)},${140+Math.floor(Math.random()*40)},${130+Math.floor(Math.random()*30)},`
    };
    case 'acid': return {
      x: Math.random()*W, y: -10,
      w: 0.5 + Math.random()*1.2, h: 8 + Math.random()*18,
      op: 0.06 + Math.random()*0.18, sp: 4 + Math.random()*6,
      color: Math.random() > 0.5 ?
        `rgba(${60+Math.floor(Math.random()*60)},${200+Math.floor(Math.random()*55)},${80+Math.floor(Math.random()*60)},` :
        `rgba(${120+Math.floor(Math.random()*80)},${240+Math.floor(Math.random()*15)},${40+Math.floor(Math.random()*40)},`
    };
    case 'rain': return {
      x: Math.random()*W, y: Math.random()*H - H,
      w: 0.8 + Math.random()*0.8, h: 14 + Math.random()*22,
      op: 0.08 + Math.random()*0.18, sp: 10 + Math.random()*8,
      angle: 0.18 + Math.random()*0.1, color: 'rgba(120,180,255,'
    };
    case 'embers': return {
      x: Math.random()*W, y: H + Math.random()*50,
      r: 0.5 + Math.random()*2, op: 0.1 + Math.random()*0.5,
      sp: 0.3 + Math.random()*0.8, dr: (Math.random()-.5)*0.8,
      wb: Math.random()*Math.PI*2, life: Math.random(),
      color: Math.random()>0.5 ? 'rgba(255,120,30,' : 'rgba(255,200,50,'
    };
    case 'matrix': return {
      x: Math.floor(Math.random()*(W/10))*10, y: Math.random()*H - H,
      char: String.fromCharCode(0x30A0 + Math.floor(Math.random()*96)),
      op: 0.05 + Math.random()*0.35, sp: 1.5 + Math.random()*3.5,
      size: 10 + Math.floor(Math.random()*4),
      timer: Math.floor(Math.random()*20)
    };
    case 'stars': return {
      x: Math.random()*W, y: Math.random()*H,
      r: 0.2 + Math.random()*1.4, op: 0.1 + Math.random()*0.7,
      twinkle: Math.random()*Math.PI*2, twinkleSpeed: 0.01 + Math.random()*0.04,
      drift: (Math.random()-.5)*0.05
    };
    case 'petals': return {
      x: Math.random()*W, y: Math.random()*H - H,
      w: 4 + Math.random()*6, h: 2 + Math.random()*3,

    };
  }
}

function initParticles() {
  if(effectAnimId) { cancelAnimationFrame(effectAnimId); effectAnimId = null; }
  effectParticles = [];
  bgCtx.clearRect(0, 0, bgCanvas.width, bgCanvas.height);
  if(currentEffect === 'none') return;
  const COUNT = { snow:140, ash:90, acid:200, rain:180, embers:120, matrix:80, stars:180, petals:70, bubbles:50 }[currentEffect] || 0;
  for(let i=0;i<COUNT;i++) {
    const p = makeParticle(currentEffect);
    if(p) effectParticles.push(p);
  }
  drawParticles();
}

function drawParticles() {
  const W = bgCanvas.width, H = bgCanvas.height;
  bgCtx.clearRect(0, 0, W, H);
  for(let p of effectParticles) {
    switch(currentEffect) {
      case 'snow':
        p.wb += 0.007; p.y += p.sp; p.x += Math.sin(p.wb)*p.dr;
        if(p.y > H+4) { p.y=-4; p.x=Math.random()*W; }
        bgCtx.beginPath(); bgCtx.arc(p.x, p.y, p.r, 0, Math.PI*2);
        bgCtx.fillStyle = p.color + p.op + ')'; bgCtx.fill();
        break;
      case 'ash':
        p.wb += 0.005; p.y += p.sp; p.x += Math.sin(p.wb)*p.dr; p.rot += p.rotSpeed;
        if(p.y > H+8) { p.y=-8; p.x=Math.random()*W; }
        bgCtx.save(); bgCtx.translate(p.x, p.y); bgCtx.rotate(p.rot);
        bgCtx.fillStyle = p.color + p.op + ')';
        if(p.isRect) { bgCtx.fillRect(-p.r, -p.r*0.5, p.r*2, p.r); }
        else { bgCtx.beginPath(); bgCtx.arc(0,0,p.r,0,Math.PI*2); bgCtx.fill(); }
        bgCtx.restore(); break;
      case 'acid':
        p.y += p.sp;
        if(p.y > H+30) { p.y=-20; p.x=Math.random()*W; }
        bgCtx.fillStyle = p.color + p.op + ')'; bgCtx.fillRect(p.x, p.y, p.w, p.h);
        if(p.y+p.h > H-2 && p.y+p.h < H+10) {
          bgCtx.beginPath(); bgCtx.arc(p.x, H-2, 1.5+Math.random()*2, 0, Math.PI*2);
          bgCtx.fillStyle = p.color + (p.op*0.5) + ')'; bgCtx.fill();
        }
        break;
      case 'rain':
        p.y += p.sp; p.x -= p.angle * p.sp * 0.5;
        if(p.y > H+30) { p.y=-30; p.x=Math.random()*W; }
        bgCtx.save(); bgCtx.translate(p.x, p.y); bgCtx.rotate(-p.angle);
        bgCtx.fillStyle = p.color + p.op + ')'; bgCtx.fillRect(-p.w/2, -p.h/2, p.w, p.h);
        bgCtx.restore(); break;
      case 'embers':
        p.wb += 0.03; p.y -= p.sp; p.x += Math.sin(p.wb)*p.dr*0.5;
        p.life -= 0.003; p.op = Math.max(0, p.life * 0.6);
        if(p.y < -5 || p.life <= 0) { Object.assign(p, makeParticle('embers')); }
        bgCtx.beginPath(); bgCtx.arc(p.x, p.y, p.r, 0, Math.PI*2);
        bgCtx.fillStyle = p.color + p.op + ')'; bgCtx.fill();
        // glow
        bgCtx.beginPath(); bgCtx.arc(p.x, p.y, p.r*2.5, 0, Math.PI*2);
        bgCtx.fillStyle = p.color + (p.op*0.2) + ')'; bgCtx.fill();
        break;
      case 'matrix':
        p.y += p.sp; p.timer--;
        if(p.timer <= 0) { p.char = String.fromCharCode(0x30A0 + Math.floor(Math.random()*96)); p.timer = 5+Math.floor(Math.random()*15); }
        if(p.y > H+20) { p.y = -20; p.x = Math.floor(Math.random()*(W/10))*10; }
        bgCtx.font = p.size + 'px monospace';
        bgCtx.fillStyle = `rgba(0,255,70,${p.op})`; bgCtx.fillText(p.char, p.x, p.y);
        // bright head
        bgCtx.fillStyle = `rgba(180,255,180,${Math.min(1,p.op*2)})`; bgCtx.fillText(p.char, p.x, p.y);
        break;
      case 'stars':
        p.twinkle += p.twinkleSpeed; p.x += p.drift;
        if(p.x > W) p.x = 0; if(p.x < 0) p.x = W;
        const twinkleOp = p.op * (0.5 + 0.5*Math.sin(p.twinkle));
        bgCtx.beginPath(); bgCtx.arc(p.x, p.y, p.r, 0, Math.PI*2);
        bgCtx.fillStyle = `rgba(255,255,240,${twinkleOp})`; bgCtx.fill();
        if(p.r > 0.9) { // cross glint for bigger stars
          bgCtx.strokeStyle = `rgba(255,255,220,${twinkleOp*0.4})`; bgCtx.lineWidth = 0.5;
          bgCtx.beginPath(); bgCtx.moveTo(p.x-p.r*3,p.y); bgCtx.lineTo(p.x+p.r*3,p.y); bgCtx.stroke();
          bgCtx.beginPath(); bgCtx.moveTo(p.x,p.y-p.r*3); bgCtx.lineTo(p.x,p.y+p.r*3); bgCtx.stroke();
        }
        break;
      case 'petals':
        p.wb += 0.01; p.y += p.sp; p.x += Math.sin(p.wb)*p.dr; p.rot += p.rotSpeed;
        if(p.y > H+10) { p.y=-10; p.x=Math.random()*W; }
        bgCtx.save(); bgCtx.translate(p.x, p.y); bgCtx.rotate(p.rot);
        bgCtx.fillStyle = `hsla(${p.hue},80%,75%,${p.op})`;
        bgCtx.beginPath(); bgCtx.ellipse(0, 0, p.w, p.h, 0, 0, Math.PI*2);
        bgCtx.fill(); bgCtx.restore(); break;
      case 'bubbles':
        p.wb += 0.015; p.y -= p.sp; p.x += Math.sin(p.wb)*p.dr;
        if(p.y < -p.r*2) { Object.assign(p, makeParticle('bubbles')); }
        bgCtx.beginPath(); bgCtx.arc(p.x, p.y, p.r, 0, Math.PI*2);
        bgCtx.strokeStyle = `rgba(180,220,255,${p.op*1.5})`; bgCtx.lineWidth = 0.8; bgCtx.stroke();
        bgCtx.fillStyle = `rgba(180,220,255,${p.op*0.15})`; bgCtx.fill();
        // highlight
        bgCtx.beginPath(); bgCtx.arc(p.x - p.r*0.3, p.y - p.r*0.3, p.r*0.25, 0, Math.PI*2);
        bgCtx.fillStyle = `rgba(255,255,255,${p.op*0.8})`; bgCtx.fill();
        break;
    }
  }
  effectAnimId = requestAnimationFrame(drawParticles);
}

function applyTheme(k) {
  const t = THEMES[k] || THEMES.default;
  const r = document.documentElement.style;
  r.setProperty('--bg', t.bg);
  r.setProperty('--bg2', t.bg2);
  r.setProperty('--bg3', t.bg3);
  r.setProperty('--bg4', t.bg4);
  r.setProperty('--accent', t.accent);
  r.setProperty('--accent2', t.accent2);
  r.setProperty('--accent3', t.accent3);
  r.setProperty('--text', t.text);
  r.setProperty('--muted', t.muted);
  r.setProperty('--muted2', t.muted2);
  r.setProperty('--border', 'rgba(255,255,255,0.055)');
  r.setProperty('--border2', 'rgba(255,255,255,0.11)');
  document.body.style.background = t.bg;
  state.theme = k;
  persist();
  renderThemeGrid();
  renderObThemeGrid();
  // Update dynamic sidebar if no custom bg color override
  if(!state.dynSidebarColor) applyDefaultSidebar();
}

function removeFromLibraryByIdx(idx) {

/* ════ BOOT ════ */
function boot() {
  restore();
  applyTheme(state.theme);
  document.getElementById('panicDisp').textContent = state.panicKey;
  document.getElementById('panicUrlInput').value = state.panicUrl;
  // Restore wallpaper
  if(state.customBg) {
    applyWallpaper(state.customBg);
  }

  const bar = document.getElementById('ld-bar'), status = document.getElementById('ld-status');
  let p=0, si=0;
  const iv = setInterval(()=>{
    p += 5 + Math.random()*12; if(p>100) p=100;
    bar.style.width = p+'%';
    if(Math.round(p)%22===0 && si<STATUS_MSGS.length) status.textContent = STATUS_MSGS[si++];
    if(p>=100) {
      clearInterval(iv); status.textContent = 'ready.';
      setTimeout(()=>{
        if(state.onboarded) {
          if(state.experience==='full') document.getElementById('navProxy').style.display='';
          document.getElementById('onboarding').classList.add('hide');
          renderAll();
          startEffect();
        } else { goStep(2); }
      }, 420);
    }
  }, 90);
}

}

// Collect ALL localStorage keys (everything, not just syn4)
function captureLocalStorage() {
  const out = {};
  for(let i=0;i<localStorage.length;i++) {
    const k = localStorage.key(i);
    try { out[k] = localStorage.getItem(k); } catch(e){}
  }
  return out;
}

// Restore all localStorage from snapshot
function restoreLocalStorage(snap) {
  if(!snap) return;
  for(const [k,v] of Object.entries(snap)) {
    try { localStorage.setItem(k, v); } catch(e){}
  }
}

// Dump all IndexedDB databases we can access
async function captureIndexedDB() {
  const out = {};
  try {
    const dbs = await indexedDB.databases();
    for(const dbInfo of dbs) {
      try {
        const dbData = await dumpIDB(dbInfo.name);
        if(dbData) out[dbInfo.name] = dbData;
      } catch(e){}
    }
  } catch(e){}
  return out;
}

function dumpIDB(dbName) {
  return new Promise((resolve) => {
    const req = indexedDB.open(dbName);
    req.onerror = () => resolve(null);
    req.onsuccess = ev => {
      const db = ev.target.result;
      const result = { version: db.version, stores: {} };
      const storeNames = Array.from(db.objectStoreNames);
      if(!storeNames.length) { db.close(); return resolve(result); }
      let done = 0;
      const tx = db.transaction(storeNames, 'readonly');
      storeNames.forEach(sName => {
        const store = tx.objectStore(sName);
        const items = [];
        const cursor = store.openCursor();
        cursor.onsuccess = ev2 => {
          const c = ev2.target.result;
          if(c) { try { items.push({ key: c.key, value: c.value }); } catch(e){} c.continue(); }
          else {
            result.stores[sName] = { keyPath: store.keyPath, autoIncrement: store.autoIncrement, items };
            done++;
            if(done === storeNames.length) { db.close(); resolve(result); }
          }
        };
        cursor.onerror = () => { done++; if(done===storeNames.length){db.close();resolve(result);} };
      });
      tx.onerror = () => { db.close(); resolve(result); };
    };
  });
}

async function restoreIndexedDB(idbSnap) {
  if(!idbSnap) return;
  for(const [dbName, dbData] of Object.entries(idbSnap)) {
    try { await restoreIDB(dbName, dbData); } catch(e){}
  }
}

function restoreIDB(dbName, dbData) {
  return new Promise((resolve) => {
    const req = indexedDB.open(dbName, dbData.version);
    req.onupgradeneeded = ev => {
      const db = ev.target.result;
      for(const [sName, sData] of Object.entries(dbData.stores)) {
        if(!db.objectStoreNames.contains(sName)) {
          try {
            db.createObjectStore(sName, {
              keyPath: sData.keyPath || undefined,
              autoIncrement: sData.autoIncrement
            });
          } catch(e){}
        }
      }
    };
    req.onsuccess = ev => {
      const db = ev.target.result;
      const storeNames = Object.keys(dbData.stores).filter(s => db.objectStoreNames.contains(s));
      if(!storeNames.length) { db.close(); return resolve(); }
      const tx = db.transaction(storeNames, 'readwrite');
      let pending = storeNames.length;
      storeNames.forEach(sName => {
        const store = tx.objectStore(sName);
        const items = dbData.stores[sName].items || [];
        let iDone = 0;
        if(!items.length) { pending--; if(!pending){db.close();resolve();} return; }
        items.forEach(item => {
          try {
            const r = store.keyPath ? store.put(item.value) : store.put(item.value, item.key);
            r.onsuccess = r.onerror = () => { iDone++; if(iDone===items.length){pending--;if(!pending){db.close();resolve();}} };
          } catch(e) { iDone++; if(iDone===items.length){pending--;if(!pending){db.close();resolve();}} }
        });
      });
      tx.onerror = () => { db.close(); resolve(); };
    };
    req.onerror = () => resolve();
  });
}

async function loadData(){
  const input = document.createElement('input');
  input.type = 'file'; input.accept = '.json';
  input.onchange = async e => {
    const f = e.target.files[0]; if(!f) return;
    const text = await f.text();
    try {
      const parsed = JSON.parse(text);
      // Detect bundle vs legacy plain state file
      if(parsed._type === 'synapse_bundle_v1') {
        // Restore synapse state
        Object.assign(state, parsed.synapseState);
        if(!state.library) state.library = [];
        // Restore localStorage (but write back after, so syn4 key gets the updated state)
        if(parsed.localStorage) restoreLocalStorage(parsed.localStorage);
        // Restore IndexedDB async
        if(parsed.indexedDB) {
          showToast('Restoring game saves...');
          await restoreIndexedDB(parsed.indexedDB);
        }
      } else {
        // Legacy: plain state object
        Object.assign(state, parsed);
        if(!state.library) state.library = [];
      }
      persist();
      applyTheme(state.theme);
      document.getElementById('panicDisp').textContent = state.panicKey;
      document.getElementById('panicUrlInput').value = state.panicUrl;
      if(state.experience==='full') document.getElementById('navProxy').style.display='';
      if(state.customBg) applyWallpaper(state.customBg);
      const tds = document.getElementById('toggleDynSidebar');
      if(tds) tds.checked = !!state.dynSidebar;
      const tdt = document.getElementById('toggleDynTheme');
      if(tdt) tdt.checked = !!state.dynTheme;
      const te = document.getElementById('toggleEffect');
      if(te) te.checked = !!state.effectEnabled;
      state.onboarded = true; persist();
      document.getElementById('onboarding').classList.add('hide');
      renderAll(); startEffect();
      showToast('✓ Data loaded!');
    } catch(err) {
      showToast('⚠ Could not read file');
    }
  };
  input.click();
}

/* ════ DYNAMIC THEME FROM IMAGE ════ */
function sampleImageColors(imgEl) {
  // Sample a 40x40 thumbnail for speed
  const c = document.createElement('canvas');
  c.width = 40; c.height = 40;
  const ctx2 = c.getContext('2d');
  try {
    ctx2.drawImage(imgEl, 0, 0, 40, 40);
    const d = ctx2.getImageData(0, 0, 40, 40).data;
    // Build a rough palette by averaging quadrants
    const quads = [{ r:0,g:0,b:0,n:0 },{ r:0,g:0,b:0,n:0 },{ r:0,g:0,b:0,n:0 },{ r:0,g:0,b:0,n:0 }];
    for(let y=0;y<40;y++) for(let x=0;x<40;x++) {
      const qi = (y<20?0:2)+(x<20?0:1);
      const i=(y*40+x)*4;
      quads[qi].r+=d[i]; quads[qi].g+=d[i+1]; quads[qi].b+=d[i+2]; quads[qi].n++;
    }
    const avg = quads.map(q=>({r:Math.round(q.r/q.n),g:Math.round(q.g/q.n),b:Math.round(q.b/q.n)}));
    // Overall average
    const all = {r:0,g:0,b:0};
    avg.forEach(q=>{all.r+=q.r;all.g+=q.g;all.b+=q.b;});
    all.r=Math.round(all.r/4); all.g=Math.round(all.g/4); all.b=Math.round(all.b/4);
    // Find most saturated/vivid quadrant for accent
    let bestSat=0, accent=avg[0];
    avg.forEach(q=>{
      const max=Math.max(q.r,q.g,q.b), min=Math.min(q.r,q.g,q.b);
      const sat = max>0?(max-min)/max:0;
      if(sat>bestSat){bestSat=sat;accent=q;}
    });
    return { avg: all, accent, quads: avg };
  } catch(e) { return null; }
}

function rgbToHex(r,g,b){return'#'+[r,g,b].map(v=>v.toString(16).padStart(2,'0')).join('');}

function applyDynThemeFromImg(imgEl) {
  const colors = sampleImageColors(imgEl);
  if(!colors) return;
  const { avg, accent } = colors;

  // Build dark bg variants from the average color (keep dark)
  const darken = (v,f) => Math.min(255,Math.max(0,Math.round(v*f)));
  const mix = (a,b,t) => Math.round(a+(b-a)*t);

  const bg   = rgbToHex(darken(avg.r,.08), darken(avg.g,.08), darken(avg.b,.08));
  const bg2  = rgbToHex(darken(avg.r,.13), darken(avg.g,.13), darken(avg.b,.13));
  const bg3  = rgbToHex(darken(avg.r,.20), darken(avg.g,.20), darken(avg.b,.20));
  const bg4  = rgbToHex(darken(avg.r,.27), darken(avg.g,.27), darken(avg.b,.27));

  // Boost accent saturation
  const aMax = Math.max(accent.r,accent.g,accent.b);
  const boost = aMax > 0 ? Math.min(255/aMax, 1.6) : 1;
  const ar = Math.min(255,Math.round(accent.r*boost));
  const ag = Math.min(255,Math.round(accent.g*boost));
  const ab = Math.min(255,Math.round(accent.b*boost));
  const accentHex  = rgbToHex(ar,ag,ab);
  const accent2Hex = rgbToHex(mix(ar,255,.28),mix(ag,255,.28),mix(ab,255,.28));
  const accent3Hex = rgbToHex(mix(ar,255,.52),mix(ag,255,.52),mix(ab,255,.52));

  // Text: high contrast against bg
  const lum = 0.299*avg.r + 0.587*avg.g + 0.114*avg.b;
  const text   = lum < 100 ? '#eeeef3' : '#0a0a0f';
  const muted  = lum < 100 ? rgbToHex(darken(avg.r,.45),darken(avg.g,.45),darken(avg.b,.45)) : '#6b7280';
  const muted2 = lum < 100 ? rgbToHex(darken(avg.r,.65),darken(avg.g,.65),darken(avg.b,.65)) : '#9ca3af';

  const r = document.documentElement.style;
  r.setProperty('--bg',   bg);
  r.setProperty('--bg2',  bg2);
  r.setProperty('--bg3',  bg3);
  r.setProperty('--bg4',  bg4);
  r.setProperty('--accent',  accentHex);
  r.setProperty('--accent2', accent2Hex);

function extractDominantColor(imgEl, cb) {
  const c = document.createElement('canvas');
  c.width = 50; c.height = 50;
  const ctx2 = c.getContext('2d');
  try {
    ctx2.drawImage(imgEl, 0, 0, 50, 50);
    const d = ctx2.getImageData(0, 0, 50, 50).data;
    let r=0,g=0,b=0,count=0;
    for(let i=0;i<d.length;i+=4*5){r+=d[i];g+=d[i+1];b+=d[i+2];count++;}
    r=Math.floor(r/count); g=Math.floor(g/count); b=Math.floor(b/count);
    cb(r,g,b);
  } catch(e) { cb(14,14,19); }
}

  const s = document.getElementById('sidebar');
  s.style.removeProperty('background');
  s.style.removeProperty('border-color');
  s.style.removeProperty('backdrop-filter');

    const dk = (v,a) => Math.max(0,Math.floor(v*a));
    const sidebar = document.getElementById('sidebar');
    sidebar.style.background = `rgba(${dk(r,.28)},${dk(g,.28)},${dk(b,.28)},0.88)`;
    sidebar.style.borderColor = `rgba(${r},${g},${b},0.12)`;
    sidebar.style.backdropFilter = 'blur(24px) saturate(1.6)';
  });
}

  lost: 0,
  bonusesClaimed: 0,

function casinoPersist() {
  state.casino = casino;
  persist();
}

function casinoRestore() {
  if(state.casino) Object.assign(casino, state.casino);
}

function casinoFmt(n) {
  return '$' + Math.abs(n).toFixed(2);
}

function updateBalanceUI() {
  const el = document.getElementById('casinoBalanceDisp');
  const sub = document.getElementById('casinoWageredDisp');
  if(el) el.textContent = '$' + casino.balance.toFixed(2);
  if(sub) sub.textContent = 'wagered: $' + casino.wagered.toFixed(2);
  // Check bonus eligibility
  const bb = document.getElementById('bonusBanner');
  if(bb) bb.style.display = casino.balance < 0.5 ? '' : 'none';
}

function casinoWin(amount, game, bet) {
  const profit = amount - bet;
  casino.balance += amount;
  casino.won += amount;
  casino.totalGames++;
  if(profit > casino.biggestWin) casino.biggestWin = profit;
  casino.history.unshift({game, bet, result:'win', profit, ts: Date.now()});
  if(casino.history.length > 200) casino.history.pop();
  casinoPersist(); updateBalanceUI();
  // Flash
  const f = document.createElement('div'); f.className = 'win-flash';
  document.body.appendChild(f); setTimeout(()=>f.remove(), 600);
}

function casinoLose(bet, game) {
  casino.balance -= bet;
  casino.wagered += bet;
  casino.sessionWagered += bet;
  casino.lost += bet;
  casino.totalGames++;
  if(bet > casino.biggestLoss) casino.biggestLoss = bet;
  casino.history.unshift({game, bet, result:'loss', profit:-bet, ts: Date.now()});
  if(casino.history.length > 200) casino.history.pop();
  casinoPersist(); updateBalanceUI();
  const f = document.createElement('div'); f.className = 'loss-flash';
  document.body.appendChild(f); setTimeout(()=>f.remove(), 500);
}

function casinoBet(bet, game) {
  // just track the wager
  casino.wagered += bet;
  casino.sessionWagered += bet;
  casinoPersist(); updateBalanceUI();
}

function claimBonus() {
  casino.balance += 50;
  casino.bonusesClaimed++;
  casinoPersist(); updateBalanceUI();
  showToast('🎁 $50 bonus claimed!');
  renderStats();
}

function resetCasino() {
  casino = { balance:200, wagered:0, won:0, lost:0, bonusesClaimed:0, totalGames:0, biggestWin:0, biggestLoss:0, sessionWagered:0, history:[] };
  casinoPersist(); updateBalanceUI(); renderStats();

}

function weightedRoll(pool) {
  const totalW = pool.reduce((s,e)=>s+e.w, 0);
  let r = Math.random() * totalW;
  for(const entry of pool) { r -= entry.w; if(r <= 0) return entry.item; }
  return pool[pool.length-1].item;
}

let caseOpenerActive = false;

function openCaseModal(caseId) {
  if(caseOpenerActive) return;
  const c = CASES.find(x=>x.id===caseId); if(!c) return;
  if(casino.balance < c.price) { showToast('Insufficient balance!'); switchCasino('stats'); return; }

  caseOpenerActive = true;
  // Build the strip items (50 visible, result at center ~25)
  const STRIP_COUNT = 60;
  const items = [];
  for(let i=0;i<STRIP_COUNT;i++) items.push(weightedRoll(c.pool));
  // Force result at position 28
  const resultKey = weightedRoll(c.pool);
  items[28] = resultKey;

  const ITEM_W = 126; // px including gap

  const modal = document.createElement('div');
  modal.className = 'case-opener-modal';
  modal.innerHTML = `
    <div style="font-size:20px;font-weight:700;letter-spacing:-.3px;margin-bottom:4px;">${c.emoji} Opening ${c.name}</div>
    <div class="case-spinner-wrap">
      <div class="case-spinner-inner" id="caseStrip">
        ${items.map(key=>{
          const item = CASE_ITEMS[key];
          const rc = {common:'r-common',uncommon:'r-uncommon',rare:'r-rare',epic:'r-epic',legendary:'r-legendary',jackpot:'r-jackpot'}[item.rarity]||'r-common';
          return `<div class="case-item ${rc}"><div class="case-item-emoji">${item.emoji}</div><div class="case-item-name">${item.name}</div><div class="case-item-value">${casinoFmt(item.value)}</div></div>`;
        }).join('')}
      </div>
      <div class="case-pointer"></div>
    </div>
    <div id="caseResultWrap" style="display:none;"></div>
    <button class="casino-btn danger" id="caseCloseBtn" style="display:none;" onclick="document.body.querySelector('.case-opener-modal').remove();caseOpenerActive=false;">Close</button>
  `;
  document.body.appendChild(modal);

  // Deduct cost
  casino.balance -= c.price;
  casino.wagered += c.price;
  casino.sessionWagered += c.price;
  updateBalanceUI();

  // Animate
  const strip = modal.querySelector('#caseStrip');
  const targetOffset = -(28 * ITEM_W) + (modal.querySelector('.case-spinner-wrap').offsetWidth / 2) - ITEM_W/2;
  let pos = 0;
  let vel = 35;
  const target = targetOffset;
  let animId;

  function animateStrip() {
    const dist = target - pos;
    if(Math.abs(dist) < 2 && Math.abs(vel) < 1) {
      pos = target;
      strip.style.transform = `translateX(${pos}px)`;
      showCaseResult(resultKey, c, modal);
      return;
    }
    vel += dist * 0.04;
    vel *= 0.88;
    pos += vel;
    strip.style.transform = `translateX(${pos}px)`;
    animId = requestAnimationFrame(animateStrip);
  }
  setTimeout(() => animateStrip(), 100);
}

  if(c.price > casino.biggestLoss) casino.biggestLoss = c.price;
  casino.history.unshift({game:'cases', bet:c.price, result: profit>=0?'win':'loss', profit, ts:Date.now()});
  if(casino.history.length>200) casino.history.pop();

function startBattle() {

  const rounds = parseInt(document.getElementById('battleRounds').value) || 3;
  const c = CASES.find(x=>x.id===caseId); if(!c) return;
  const totalCost = c.price * rounds;
  if(casino.balance < totalCost) { showToast('Need ' + casinoFmt(totalCost) + ' for this battle!'); return; }

  casino.balance -= totalCost;
  casino.wagered += totalCost;
  casino.sessionWagered += totalCost;

    const pk = weightedRoll(c.pool); const pItem = CASE_ITEMS[pk];

    casino.won += winnings;

      </div>
      <div class="battle-vs">VS</div>
      <div class="battle-player ${!playerWon?'winner':'loser'}">
        <div style="font-size:13px;font-weight:600;margin-bottom:10px;">${!playerWon?'👑':'💀'} House</div>
        <div class="battle-items-won">
          ${houseItems.map(i=>`<div class="battle-item-chip r-${i.rarity}" style="background:rgba(255,255,255,.05);border:1px solid var(--border);">${i.emoji} ${casinoFmt(i.value)}</div>`).join('')}
        </div>
        <div style="font-size:16px;font-weight:700;font-family:var(--mono);margin-top:10px;color:${!playerWon?'var(--green)':'#ef4444'};">${casinoFmt(houseTotal)}</div>
      </div>

    <div style="text-align:center;font-size:18px;font-weight:700;color:${playerWon?'var(--green)':'#ef4444'};">

/* ══════════════════════════════════════
   CRASH
══════════════════════════════════════ */
let crashState = 'waiting'; // waiting | running | crashed
let crashMultiplier = 1.0;
let crashTarget = 1.0;
let crashAnimId = null;
let crashBetAmount = 0;
let crashBetPlaced = false;
let crashCashedOut = false;
let crashHistory = [];
let crashCanvas, crashCtx, crashPoints = [];

function initCrashCanvas() {
  crashCanvas = document.getElementById('crashCanvas'); if(!crashCanvas) return;
  const rect = crashCanvas.parentElement.getBoundingClientRect();
  crashCanvas.width = rect.width || 600;
  crashCanvas.height = rect.height || 260;
  crashCtx = crashCanvas.getContext('2d');
}

function generateCrashTarget() {
  // House edge ~4%
  const r = Math.random();
  if(r < 0.04) return 1.0; // instant crash
  return Math.max(1.01, 0.99 / (1 - Math.random()));
}

function startCrashRound() {
  crashState = 'running';
  crashMultiplier = 1.0;
  crashTarget = generateCrashTarget();
  crashPoints = [[0, 1.0]];
  crashCashedOut = false;
  let tick = 0;

  const multEl = document.getElementById('crashMult');
  const statusEl = document.getElementById('crashStatus');
  const cashoutBtn = document.getElementById('crashCashoutBtn');
  const betBtn = document.getElementById('crashBetBtn');
  if(multEl) { multEl.textContent = '1.00×'; multEl.classList.remove('crashed'); multEl.style.color = 'var(--green)'; }
  if(statusEl) statusEl.textContent = 'in progress...';
  if(cashoutBtn && crashBetPlaced) cashoutBtn.disabled = false;
  if(betBtn) betBtn.disabled = true;

  function tick_fn() {
    tick++;
    crashMultiplier = Math.pow(1.0028, tick);
    if(multEl) multEl.textContent = crashMultiplier.toFixed(2) + '×';
    crashPoints.push([tick, crashMultiplier]);
    drawCrashChart();

    // Color changes
    if(crashMultiplier > 3) { if(multEl) multEl.style.color = '#f59e0b'; }
    if(crashMultiplier > 7) { if(multEl) multEl.style.color = '#ef4444'; }

    if(crashMultiplier >= crashTarget) {
      crashState = 'crashed';
      if(multEl) { multEl.classList.add('crashed'); multEl.style.color = '#ef4444'; }
      if(statusEl) statusEl.textContent = '💥 CRASHED!';
      if(cashoutBtn) cashoutBtn.disabled = true;
      if(betBtn) betBtn.disabled = false;

      // Record to history
      crashHistory.unshift(crashMultiplier.toFixed(2));
      if(crashHistory.length > 12) crashHistory.pop();
      renderCrashHistory();

      // If bet was placed and not cashed out — lose
      if(crashBetPlaced && !crashCashedOut) {
        casino.lost += crashBetAmount;
        casino.totalGames++;
        casino.history.unshift({game:'crash', bet:crashBetAmount, result:'loss', profit:-crashBetAmount, ts:Date.now()});
        if(casino.history.length>200) casino.history.pop();
        casinoPersist(); updateBalanceUI();
        const bsEl = document.getElementById('crashBetStatus');
        if(bsEl) bsEl.innerHTML = `<span style="color:#ef4444;">💥 Crashed at ${crashMultiplier.toFixed(2)}× — Lost ${casinoFmt(crashBetAmount)}</span>`;
      }
      crashBetPlaced = false;

      // Queue next round
      setTimeout(() => {
        if(statusEl) statusEl.textContent = 'starting in 3s...';
        setTimeout(startCrashRound, 3000);
      }, 1500);
      return;
    }
    crashAnimId = requestAnimationFrame(tick_fn);
  }
  crashAnimId = requestAnimationFrame(tick_fn);
}

function drawCrashChart() {
  if(!crashCtx) return;
  const W = crashCanvas.width, H = crashCanvas.height;
  crashCtx.clearRect(0,0,W,H);

  if(crashPoints.length < 2) return;
  const maxTick = Math.max(100, crashPoints[crashPoints.length-1][0]);
  const maxMult = Math.max(2, crashPoints[crashPoints.length-1][1] * 1.2);

  const toX = t => (t/maxTick) * (W-40) + 20;
  const toY = m => H - 20 - ((m-1)/(maxMult-1)) * (H-40);

  crashCtx.strokeStyle = crashState==='crashed' ? 'rgba(239,68,68,0.6)' : 'rgba(34,197,94,0.5)';
  crashCtx.lineWidth = 2.5;
  crashCtx.beginPath();
  crashPoints.forEach(([t,m],i) => {
    const x = toX(t), y = toY(m);
    i===0 ? crashCtx.moveTo(x,y) : crashCtx.lineTo(x,y);
  });
  crashCtx.stroke();

  // Fill
  crashCtx.fillStyle = crashState==='crashed' ? 'rgba(239,68,68,0.07)' : 'rgba(34,197,94,0.07)';
  crashCtx.lineTo(toX(crashPoints[crashPoints.length-1][0]), H-20);
  crashCtx.lineTo(toX(0), H-20);
  crashCtx.closePath();
  crashCtx.fill();
}

function renderCrashHistory() {
  const el = document.getElementById('crashHistory'); if(!el) return;
  el.innerHTML = crashHistory.map(m => {
    const v = parseFloat(m);
    const cls = v >= 3 ? 'high' : v >= 1.5 ? 'mid' : 'low';
    return `<div class="crash-pill ${cls}">${m}×</div>`;
  }).join('');
}

function placeCrashBet() {
  if(crashState !== 'waiting' && crashState !== 'running') return;
  if(crashBetPlaced) { showToast('Bet already placed'); return; }
  const bet = parseFloat(document.getElementById('crashBet').value) || 5;
  if(bet <= 0) return;
  if(casino.balance < bet) { showToast('Insufficient balance!'); return; }
  casino.balance -= bet;
  casino.wagered += bet;
  casino.sessionWagered += bet;
  crashBetAmount = bet;
  crashBetPlaced = true;
  updateBalanceUI();
  const bsEl = document.getElementById('crashBetStatus');

  showToast('Bet placed: ' + casinoFmt(bet));
}

function cashoutCrash() {
  if(!crashBetPlaced || crashCashedOut || crashState !== 'running') return;
  crashCashedOut = true;
  const payout = crashBetAmount * crashMultiplier;
  const profit = payout - crashBetAmount;
  casino.balance += payout;
  casino.won += payout;
  casino.totalGames++;
  if(profit > casino.biggestWin) casino.biggestWin = profit;
  casino.history.unshift({game:'crash', bet:crashBetAmount, result:'win', profit, ts:Date.now()});
  if(casino.history.length>200) casino.history.pop();
  casinoPersist(); updateBalanceUI();
  const bsEl = document.getElementById('crashBetStatus');
  if(bsEl) bsEl.innerHTML = `<span style="color:var(--green);">✓ Cashed out at ${crashMultiplier.toFixed(2)}× — Won ${casinoFmt(payout)}</span>`;
  document.getElementById('crashCashoutBtn').disabled = true;
  showToast('💰 Cashed out: +' + casinoFmt(profit));
}

const SLOT_SYMBOLS = ['🍒','🍋','🍊','🍇','💎','7️⃣','⭐','🎰','🔔','🍀'];
const SLOT_WEIGHTS = [20,18,15,12,8,5,4,3,2,1]; // weighted
const SLOT_PAYOUTS = {
  '🍒🍒🍒':2,'🍋🍋🍋':3,'🍊🍊🍊':4,'🍇🍇🍇':5,

function initSlots() {
  for(let r=0;r<3;r++) {

function weightedSlotSym() {
  const total = SLOT_WEIGHTS.reduce((a,b)=>a+b,0);
  let r = Math.random()*total;
  for(let i=0;i<SLOT_SYMBOLS.length;i++) { r -= SLOT_WEIGHTS[i]; if(r<=0) return SLOT_SYMBOLS[i]; }
  return SLOT_SYMBOLS[0];
}

function spinSlots() {
  if(slotsSpinning) return;
  const bet = parseFloat(document.getElementById('slotsBet').value) || 2;
  if(bet <= 0) return;
  if(casino.balance < bet) { showToast('Insufficient balance!'); return; }

  slotsSpinning = true;
  document.getElementById('slotsBtn').disabled = true;
  document.getElementById('slotsWin').textContent = '';
  casino.balance -= bet;
  casino.wagered += bet;
  casino.sessionWagered += bet;
  updateBalanceUI();

  const results = [weightedSlotSym(), weightedSlotSym(), weightedSlotSym()];
  const SPIN_TICKS = [22, 28, 34]; // different stop times per reel

  for(let r=0;r<3;r++) {
    const reel = document.getElementById('reel'+r); if(!reel) continue;
    // Append result symbol + extras
    const extras = [...Array(SPIN_TICKS[r])].map(()=>`<div class="reel-symbol">${weightedSlotSym()}</div>`).join('');
    const resultSym = `<div class="reel-symbol">${results[r]}</div>`;
    reel.innerHTML = reel.innerHTML + extras + resultSym + [...Array(5)].map(()=>`<div class="reel-symbol">${weightedSlotSym()}</div>`).join('');

    const totalItems = reel.children.length;
    const targetIndex = totalItems - 6; // result symbol position
    let curIdx = 0; let speed = 3;

    (function spinReel(reelEl, targetIdx, rIdx) {
      function step() {
        if(curIdx >= targetIdx) {
          reelEl.style.transform = `translateY(-${targetIdx * 90}px)`;
          if(rIdx === 2) finishSpin(results, bet);
          return;
        }
        const slowZone = targetIdx - curIdx < 8;
        if(slowZone) speed = Math.max(0.5, speed * 0.85);
        curIdx += speed;
        reelEl.style.transform = `translateY(-${Math.min(curIdx, targetIdx) * 90}px)`;
        requestAnimationFrame(step);
      }
      step();
    })(reel, targetIndex, r);
  }
}

function finishSpin(results, bet) {
  slotsSpinning = false;
  document.getElementById('slotsBtn').disabled = false;

  // Check wins
  const key3 = results.join('');
  const key2a = results[0]+results[1]+'_';
  let mult = SLOT_PAYOUTS[key3] || 0;
  if(!mult && results[0]===results[1]) mult = SLOT_PAYOUTS[key2a] || 0;

  const winEl = document.getElementById('slotsWin');
  if(mult > 0) {
    const payout = bet * mult;
    const profit = payout - bet;
    casino.balance += payout;
    casino.won += payout;
    casino.totalGames++;
    if(profit > casino.biggestWin) casino.biggestWin = profit;
    casino.history.unshift({game:'slots', bet, result:'win', profit, ts:Date.now()});
    if(casino.history.length>200) casino.history.pop();
    casinoPersist(); updateBalanceUI();
    winEl.innerHTML = `<span style="color:var(--green);">+${casinoFmt(profit)} (${mult}×)</span>`;
    showToast('🎰 Win! +' + casinoFmt(profit));
  } else {
    casino.lost += bet;
    casino.totalGames++;
    casino.history.unshift({game:'slots', bet, result:'loss', profit:-bet, ts:Date.now()});
    if(casino.history.length>200) casino.history.pop();
    casinoPersist(); updateBalanceUI();
    winEl.innerHTML = `<span style="color:#ef4444;">-${casinoFmt(bet)}</span>`;
  }
}

let cupsState = 'idle'; // idle | showing | shuffling | guessing | result
let cupsBallPos = 0;
let cupsCount = 3;
let cupsAnimating = false;

function setCupsDiff(d) {
  [1,2,3].forEach(i => {
    const b = document.getElementById('cdiff'+i);
    if(b) b.style.borderColor = i===d ? 'var(--green)' : '';
    if(b) b.style.color = i===d ? 'var(--green)' : '';
  });
  cupsCount = d===1?3 : d===2?4 : 5;
  renderCups();
}

function renderCups() {
  const row = document.getElementById('cupsRow'); if(!row) return;
  row.innerHTML = '';
  for(let i=0;i<cupsCount;i++) {
    const wrap = document.createElement('div');
    wrap.className = 'cup-wrap';
    wrap.id = 'cup'+i;
    wrap.innerHTML = `
      <svg class="cup-svg" width="80" height="90" viewBox="0 0 80 90">
        <defs><linearGradient id="cg${i}" x1="0%" y1="0%" x2="100%" y2="100%">
          <stop offset="0%" style="stop-color:#7c6bff;stop-opacity:1"/>
          <stop offset="100%" style="stop-color:#a78bfa;stop-opacity:1"/>
        </linearGradient></defs>
        <path d="M10 5 L70 5 L62 75 Q60 85 40 85 Q20 85 18 75 Z" fill="url(#cg${i})" stroke="rgba(255,255,255,.15)" stroke-width="1.5"/>
        <path d="M10 5 L70 5 L65 20 L15 20 Z" fill="rgba(255,255,255,.12)"/>
      </svg>
      <div class="cup-ball hidden" id="ball${i}" style="background:radial-gradient(circle at 35% 35%,#fff8,#ef4444);"></div>
    `;
    if(cupsState === 'guessing') {
      wrap.onclick = () => guessCup(i);
      wrap.style.cursor = 'pointer';
    }
    row.appendChild(wrap);
  }
}

function startCups() {
  if(cupsAnimating) return;
  const bet = parseFloat(document.getElementById('cupsBet').value) || 5;
  if(bet <= 0) return;
  if(casino.balance < bet) { showToast('Insufficient balance!'); return; }
  casino.balance -= bet;
  casino.wagered += bet;
  casino.sessionWagered += bet;
  updateBalanceUI();

  cupsState = 'showing';
  cupsAnimating = true;
  cupsBallPos = Math.floor(Math.random() * cupsCount);
  renderCups();
  document.getElementById('cupsMsg').textContent = 'Watch the ball...';
  document.getElementById('cupsStartBtn').disabled = true;

  // Show ball
  const ballEl = document.getElementById('ball' + cupsBallPos);
  if(ballEl) ballEl.classList.remove('hidden');

  // After 1.5s, shuffle
  setTimeout(() => {
    if(ballEl) ballEl.classList.add('hidden');
    shuffleCups(bet, 0, 6 + cupsCount * 2);
  }, 1500);
}

function shuffleCups(bet, step, totalSteps) {
  if(step >= totalSteps) {
    cupsState = 'guessing';
    cupsAnimating = false;
    renderCups();
    document.getElementById('cupsMsg').textContent = 'Which cup? Pick one!';
    // Store bet for result
    window._cupsBet = bet;
    return;
  }
  // Swap two random cups
  let a = Math.floor(Math.random()*cupsCount);
  let b;
  do { b = Math.floor(Math.random()*cupsCount); } while(b===a);
  // Swap ball position if needed
  if(cupsBallPos===a) cupsBallPos=b;
  else if(cupsBallPos===b) cupsBallPos=a;

  // Animate cups sliding
  const capA = document.getElementById('cup'+a)?.querySelector('.cup-svg');
  const capB = document.getElementById('cup'+b)?.querySelector('.cup-svg');
  if(capA) { capA.style.transition='transform .18s ease'; capA.style.transform='translateX(20px)'; }
  if(capB) { capB.style.transition='transform .18s ease'; capB.style.transform='translateX(-20px)'; }
  setTimeout(()=>{
    if(capA) capA.style.transform='';
    if(capB) capB.style.transform='';
    setTimeout(()=>shuffleCups(bet, step+1, totalSteps), 60);
  }, 180);
}

function guessCup(i) {
  if(cupsState !== 'guessing') return;
  cupsState = 'result';
  const bet = window._cupsBet || 5;
  const won = i === cupsBallPos;

  // Reveal ball
  const ballEl = document.getElementById('ball' + cupsBallPos);
  if(ballEl) ballEl.classList.remove('hidden');

  document.getElementById('cupsMsg').textContent = won ? '✅ Correct! You win!' : `❌ Wrong! Ball was under cup ${cupsBallPos+1}`;

  if(won) {
    const payout = bet * (cupsCount === 3 ? 1.9 : cupsCount === 4 ? 2.7 : 3.8);
    const profit = payout - bet;
    casino.balance += payout;
    casino.won += payout;
    casino.totalGames++;
    if(profit > casino.biggestWin) casino.biggestWin = profit;
    casino.history.unshift({game:'cups', bet, result:'win', profit, ts:Date.now()});
    showToast('🎉 Won: +' + casinoFmt(profit));
  } else {
    casino.lost += bet;
    casino.totalGames++;
    casino.history.unshift({game:'cups', bet, result:'loss', profit:-bet, ts:Date.now()});
    showToast('💀 Lost: -' + casinoFmt(bet));
  }
  if(casino.history.length>200) casino.history.pop();
  casinoPersist(); updateBalanceUI();

  setTimeout(()=>{
    cupsState='idle';
    renderCups();
    document.getElementById('cupsStartBtn').disabled = false;
    document.getElementById('cupsMsg').textContent = '';
  }, 2000);
}

let plinkoRisk = 'low';

const PLINKO_ROWS = 12;
const PLINKO_MULTIPLIERS = {
  low:  [1.5,1.2,1.1,1.0,0.8,0.5,0.8,1.0,1.1,1.2,1.5],
  med:  [5.0,2.0,1.5,1.0,0.5,0.3,0.5,1.0,1.5,2.0,5.0],
  high: [25,8,3,1.5,0.5,0.2,0.5,1.5,3,8,25],
};
const PLINKO_COLORS = {
  low:  ['#22c55e','#4ade80','#86efac','#bbf7d0','#d1fae5','#a7f3d0','#d1fae5','#bbf7d0','#86efac','#4ade80','#22c55e'],
  med:  ['#f59e0b','#fbbf24','#fcd34d','#fde68a','#a7f3d0','#6ee7b7','#a7f3d0','#fde68a','#fcd34d','#fbbf24','#f59e0b'],
  high: ['#ef4444','#f97316','#fb923c','#fbbf24','#6ee7b7','#34d399','#6ee7b7','#fbbf24','#fb923c','#f97316','#ef4444'],
};

function initPlinko() {
  plinkoCanvas = document.getElementById('plinkoCanvas'); if(!plinkoCanvas) return;
  const parent = plinkoCanvas.parentElement;
  const w = parent.offsetWidth - (180 + 20);
  plinkoCanvas.width = Math.max(400, w);
  plinkoCanvas.height = 480;
  plinkoCtx = plinkoCanvas.getContext('2d');
  plinkoBalls = [];
  if(!plinkoRunning) { plinkoRunning = true; drawPlinko(); }
  renderPlinkoMults();
}

function setPlinkoRisk(r) {
  plinkoRisk = r;
  ['low','med','high'].forEach(k => {
    const b = document.getElementById('prisk-'+k);
    if(b) {
      const col = k==='low'?'var(--green)':k==='med'?'var(--accent2)':'#ef4444';
      b.style.borderColor = k===r ? col : '';
      b.style.color = k===r ? col : '';
    }
  });
  renderPlinkoMults();
}

function getPlinkoLayout() {
  const W = plinkoCanvas.width, H = plinkoCanvas.height;
  const pins = [];
  const startY = 50;
  const rowH = (H - 120) / PLINKO_ROWS;
  for(let row=0;row<PLINKO_ROWS;row++) {
    const count = row + 3;
    const rowW = (count - 1) * 40;
    const startX = W/2 - rowW/2;
    for(let col=0;col<count;col++) {
      pins.push({ x: startX + col*40, y: startY + row*rowH });
    }
  }
  return { pins, rowH, startY };
}

function dropPlinko() {
  const bet = parseFloat(document.getElementById('plinkoBet').value) || 1;
  if(bet <= 0) return;
  if(casino.balance < bet) { showToast('Need more balance!'); return; }
  casino.balance -= bet;
  casino.wagered += bet;
  casino.sessionWagered += bet;
  updateBalanceUI();
  const W = plinkoCanvas.width;
  plinkoBalls.push({
    x: W/2 + (Math.random()-.5)*8,
    y: 20,
    vx: (Math.random()-.5)*0.5,
    vy: 0,
    r: 7,
    bet,
    landed: false,
    slot: -1,
    trail: [],
    color: PLINKO_COLORS[plinkoRisk][5] || '#7c6bff',
  });
}

function autoDropPlinko() {
  const bet = parseFloat(document.getElementById('plinkoBet').value) || 1;
  let count = 0;
  const iv = setInterval(()=>{
    if(count >= 10 || casino.balance < bet) { clearInterval(iv); return; }
    dropPlinko();
    count++;
  }, 120);
}

function drawPlinko() {
  if(!plinkoCtx || !plinkoCanvas) { plinkoRunning = false; return; }
  const W = plinkoCanvas.width, H = plinkoCanvas.height;
  plinkoCtx.clearRect(0,0,W,H);

  const { pins, rowH, startY } = getPlinkoLayout();
  const mults = PLINKO_MULTIPLIERS[plinkoRisk];
  const colors = PLINKO_COLORS[plinkoRisk];
  const SLOTS = mults.length;
  const slotW = W / SLOTS;

  // Draw slots
  for(let s=0;s<SLOTS;s++) {
    const x = s * slotW;
    const multVal = mults[s];
    plinkoCtx.fillStyle = colors[s] + '22';
    plinkoCtx.fillRect(x+2, H-55, slotW-4, 50);
    plinkoCtx.fillStyle = colors[s];
    plinkoCtx.font = 'bold 11px monospace';
    plinkoCtx.textAlign = 'center';
    plinkoCtx.fillText(multVal+'×', x + slotW/2, H-28);
  }

  // Draw pins
  pins.forEach(p => {
    plinkoCtx.beginPath();
    plinkoCtx.arc(p.x, p.y, 4, 0, Math.PI*2);
    plinkoCtx.fillStyle = 'rgba(255,255,255,0.35)';
    plinkoCtx.fill();
  });

  // Physics + draw balls
  const G = 0.28;
  plinkoBalls.forEach(ball => {
    if(ball.landed) {
      // Draw a fading landed ball
      ball.trail.push({x:ball.x,y:ball.y});
      if(ball.trail.length > 3) ball.trail.shift();
      return;
    }

    ball.vy += G;
    ball.x += ball.vx;
    ball.y += ball.vy;
    // Wall bounce
    if(ball.x < ball.r) { ball.x=ball.r; ball.vx=Math.abs(ball.vx)*0.6; }
    if(ball.x > W-ball.r) { ball.x=W-ball.r; ball.vx=-Math.abs(ball.vx)*0.6; }

    // Pin collision
    pins.forEach(p => {
      const dx = ball.x - p.x, dy = ball.y - p.y;
      const dist = Math.sqrt(dx*dx+dy*dy);
      if(dist < ball.r + 4.5) {
        const nx = dx/dist, ny = dy/dist;
        ball.x = p.x + nx*(ball.r+4.6);
        ball.y = p.y + ny*(ball.r+4.6);
        const dot = ball.vx*nx + ball.vy*ny;
        ball.vx = (ball.vx - 2*dot*nx)*0.7 + (Math.random()-.5)*0.4;
        ball.vy = (ball.vy - 2*dot*ny)*0.7;
        if(Math.abs(ball.vy) < 1) ball.vy = 1;
      }
    });

    // Land in slot
    if(ball.y >= H-60) {
      ball.landed = true;
      ball.y = H-60;
      const slot = Math.min(SLOTS-1, Math.max(0, Math.floor(ball.x / slotW)));
      ball.slot = slot;
      const mult = mults[slot];
      const payout = ball.bet * mult;
      const profit = payout - ball.bet;
      casino.balance += payout;
      casino.won += payout;
      casino.totalGames++;
      if(profit > casino.biggestWin) casino.biggestWin = profit;
      casino.history.unshift({game:'plinko', bet:ball.bet, result:profit>=0?'win':'loss', profit, ts:Date.now()});
      if(casino.history.length>200) casino.history.pop();
      casinoPersist(); updateBalanceUI();
      const lw = document.getElementById('plinkoLastWin');
      if(lw) lw.innerHTML = `<span style="color:${profit>=0?'var(--green)':'#ef4444'}">${profit>=0?'+':''}${casinoFmt(profit)} (${mult}×)</span>`;
    }

    // Trail
    ball.trail.push({x:ball.x,y:ball.y});
    if(ball.trail.length > 12) ball.trail.shift();

    // Draw trail
    for(let t=0;t<ball.trail.length-1;t++) {
      plinkoCtx.beginPath();
      plinkoCtx.moveTo(ball.trail[t].x, ball.trail[t].y);
      plinkoCtx.lineTo(ball.trail[t+1].x, ball.trail[t+1].y);
      plinkoCtx.strokeStyle = `rgba(167,139,250,${(t/ball.trail.length)*0.4})`;
      plinkoCtx.lineWidth = 2;
      plinkoCtx.stroke();
    }

    // Draw ball
    const grad = plinkoCtx.createRadialGradient(ball.x-2, ball.y-2, 1, ball.x, ball.y, ball.r);
    grad.addColorStop(0, 'rgba(255,255,255,0.9)');
    grad.addColorStop(1, ball.color);
    plinkoCtx.beginPath();
    plinkoCtx.arc(ball.x, ball.y, ball.r, 0, Math.PI*2);
    plinkoCtx.fillStyle = grad;
    plinkoCtx.fill();
  });

  // Clean up landed balls after 1.5s
  plinkoBalls = plinkoBalls.filter(b => !b.landed || (Date.now() - (b.landedAt||Date.now())) < 1200);
  plinkoBalls.forEach(b => { if(b.landed && !b.landedAt) b.landedAt = Date.now(); });

  plinkoAnimId = requestAnimationFrame(drawPlinko);
}

function renderPlinkoMults() {
  const el = document.getElementById('plinkoMultDisplay'); if(!el) return;
  const mults = PLINKO_MULTIPLIERS[plinkoRisk];
  const colors = PLINKO_COLORS[plinkoRisk];
  el.innerHTML = mults.map((m,i)=>`<div class="plinko-mult" style="color:${colors[i]};background:${colors[i]}15;">${m}×</div>`).join('');
}

function renderStats() {
  const g = document.getElementById('statsGrid'); if(!g) return;
  const netProfit = casino.won - casino.wagered;
  const winRate = casino.totalGames > 0 ? ((casino.history.filter(h=>h.result==='win').length / casino.totalGames)*100).toFixed(1) : '0.0';
  const avgBet = casino.totalGames > 0 ? (casino.wagered / casino.totalGames).toFixed(2) : '0.00';
  g.innerHTML = [
    {label:'Balance',       value:'$'+casino.balance.toFixed(2), sub:'current', color:'var(--green)'},
    {label:'Total Wagered', value:'$'+casino.wagered.toFixed(2), sub:'all time', color:'var(--accent2)'},
    {label:'Total Won',     value:'$'+casino.won.toFixed(2), sub:'payouts received', color:'#86efac'},
    {label:'Total Lost',    value:'$'+casino.lost.toFixed(2), sub:'bets lost', color:'#fca5a5'},
    {label:'Net Profit',    value:(netProfit>=0?'+':'')+'$'+Math.abs(netProfit).toFixed(2), sub:'won minus wagered', color: netProfit>=0?'var(--green)':'#ef4444'},
    {label:'Games Played',  value:casino.totalGames, sub:'total rounds', color:'var(--text)'},
    {label:'Win Rate',      value:winRate+'%', sub:'of all games', color:'var(--accent2)'},
    {label:'Biggest Win',   value:'$'+casino.biggestWin.toFixed(2), sub:'single win profit', color:'#fcd34d'},
    {label:'Biggest Loss',  value:'$'+casino.biggestLoss.toFixed(2), sub:'single bet lost', color:'#fca5a5'},
    {label:'Avg Bet',       value:'$'+avgBet, sub:'per game', color:'var(--text)'},
    {label:'Bonuses',       value:casino.bonusesClaimed, sub:'free $50 claims', color:'#fcd34d'},
    {label:'Session Wagered',value:'$'+casino.sessionWagered.toFixed(2), sub:'this session', color:'var(--muted2)'},
  ].map(s=>`<div class="stat-card">
    <div class="stat-label">${s.label}</div>
    <div class="stat-value" style="color:${s.color}">${s.value}</div>
    <div class="stat-sub">${s.sub}</div>
  </div>`).join('');

  // Recent history
  const histHtml = casino.history.slice(0,20).map(h=>{
    const t = new Date(h.ts).toLocaleTimeString();
    return `<div style="display:flex;justify-content:space-between;padding:6px 0;border-bottom:1px solid rgba(255,255,255,.04);font-size:12px;font-family:var(--mono);">
      <span style="color:var(--muted2);">${h.game}</span>
      <span style="color:var(--muted);">bet: $${h.bet.toFixed(2)}</span>
      <span style="color:${h.profit>=0?'var(--green)':'#ef4444'}">${h.profit>=0?'+':''}\$${Math.abs(h.profit).toFixed(2)}</span>
      <span style="color:var(--muted);font-size:10px;">${t}</span>
    </div>`;
  }).join('');
  if(histHtml) {
    g.innerHTML += `<div style="grid-column:1/-1;background:var(--bg2);border:1px solid var(--border);border-radius:14px;padding:16px;">
      <div class="stat-label" style="margin-bottom:10px;">Recent History</div>
      ${histHtml}
    </div>`;
  }

  // Bonus banner
  const bb = document.getElementById('bonusBanner');
  if(bb) bb.style.display = casino.balance < 0.5 ? '' : 'none';
}
