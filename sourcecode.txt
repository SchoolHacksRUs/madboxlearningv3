let activeGameUrl=null; // store url for new-tab pop-out

async function launchGame(url,title,gameIdx){
  activeGameUrl=url;
  const ov=document.createElement('div');ov.className='game-launch-ov';
  ov.innerHTML=`<div class="launch-spinner-ring"></div><div class="launch-title">${title||'Loading...'}</div><div class="launch-sub">Preparing game</div>`;
  document.body.appendChild(ov);
  try{
    const res=await fetch(url);
    const code=await res.text();
    await new Promise(r=>setTimeout(r,700));
    ov.classList.add('out');setTimeout(()=>ov.remove(),280);

    activeGameIdx=gameIdx;
    const wrap=document.getElementById('game-frame-wrap');
    const iframe=document.getElementById('game-iframe');
    wrap.classList.add('active');
    iframe.contentDocument.open();
    iframe.contentDocument.write(code);
    iframe.contentDocument.close();

    showGameControls();

  }catch(e){
    ov.classList.add('out');setTimeout(()=>ov.remove(),280);
    showToast('⚠ Failed to load game');
  }
}

/* pop current game out into a real new tab */
function gcOpenNewTab(){
  if(!activeGameUrl)return;
  const tab=window.open('','_blank');
  if(tab){
    // re-fetch and write (already have the code in iframe, but easiest to refetch)
    fetch(activeGameUrl).then(r=>r.text()).then(code=>{
      tab.document.open();tab.document.write(code);tab.document.close();
    }).catch(()=>{ tab.location.href=activeGameUrl; });
  } else {
    showToast('Popup blocked — allow popups to use this');
  }
}

/* ── collect all IndexedDB data as a plain object ── */
async function dumpIndexedDB(){
  const result={};
  const dbNames=await(indexedDB.databases?indexedDB.databases():Promise.resolve([]));
  for(const info of dbNames){
    try{
      const db=await new Promise((res,rej)=>{const r=indexedDB.open(info.name,info.version);r.onsuccess=()=>res(r.result);r.onerror=()=>rej(r.error);});
      const dbData={};
      for(const storeName of Array.from(db.objectStoreNames)){
        const tx=db.transaction(storeName,'readonly');
        const store=tx.objectStore(storeName);
        dbData[storeName]=await new Promise((res,rej)=>{
          const all=[];const req=store.openCursor();
          req.onsuccess=e=>{const cur=e.target.result;if(cur){all.push({key:cur.key,value:cur.value});cur.continue();}else res(all);};
          req.onerror=()=>rej(req.error);
        });
      }
      result[info.name]=dbData;
      db.close();
    }catch(e){/* skip dbs we can't read */}
  }
  return result;
}

/* ── restore IndexedDB from dumped object ── */
async function restoreIndexedDB(dump){
  for(const [dbName,dbData] of Object.entries(dump)){
    const storeNames=Object.keys(dbData);
    if(!storeNames.length)continue;
    try{
      const db=await new Promise((res,rej)=>{
        const r=indexedDB.open(dbName,1);
        r.onupgradeneeded=e=>{
          const d=e.target.result;
          for(const s of storeNames){if(!d.objectStoreNames.contains(s))d.createObjectStore(s);}
        };
        r.onsuccess=()=>res(r.result);r.onerror=()=>rej(r.error);
      });
      for(const [storeName,rows] of Object.entries(dbData)){
        if(!db.objectStoreNames.contains(storeName))continue;
        const tx=db.transaction(storeName,'readwrite');
        const store=tx.objectStore(storeName);
        for(const row of rows){
          try{store.put(row.value,row.key);}catch(e){}
        }
        await new Promise(res=>{tx.oncomplete=res;tx.onerror=res;});
      }
      db.close();
    }catch(e){}
  }
}

async function exportData(){
  showToast('Collecting save data…');
  try{
    /* localStorage — every key */
    const ls={};
    for(let i=0;i<localStorage.length;i++){const k=localStorage.key(i);ls[k]=localStorage.getItem(k);}

    /* cookies */
    const cookies=document.cookie;

    /* IndexedDB */
    const idb=await dumpIndexedDB();

    const full={synapseState:state,localStorage:ls,cookies,indexedDB:idb,exportedAt:new Date().toISOString()};
    const b=new Blob([JSON.stringify(full,null,2)],{type:'application/json'});
    const a=document.createElement('a');a.href=URL.createObjectURL(b);a.download='synapse_data.json';a.click();
    setTimeout(()=>URL.revokeObjectURL(a.href),1000);
    showToast('✓ Full save exported!');
  }catch(e){
    /* fallback: export just state */
    const b=new Blob([JSON.stringify(state,null,2)],{type:'application/json'});
    const a=document.createElement('a');a.href=URL.createObjectURL(b);a.download='synapse_data.json';a.click();
    setTimeout(()=>URL.revokeObjectURL(a.href),1000);
    showToast('Exported (game saves unavailable)');
  }
}

function showImportLog(parsed){
  const panel=document.getElementById('import-log');
  const list=document.getElementById('il-list');
  const sub=document.getElementById('il-sub');
  const rows=[];

  /* localStorage keys */
  if(parsed.localStorage){
    const keys=Object.keys(parsed.localStorage).filter(k=>k!=='syn4');
    if(keys.length){
      rows.push({icon:'ls',iconTxt:'LS',name:'localStorage',count:`${keys.length} key${keys.length!==1?'s':''}`});
      // also show individual notable keys
      keys.slice(0,8).forEach(k=>{
        rows.push({icon:'ls',iconTxt:'·',name:k,count:''});
      });
      if(keys.length>8)rows.push({icon:'ls',iconTxt:'…',name:`+${keys.length-8} more keys`,count:''});
    }
  }

  /* cookies */
  if(parsed.cookies&&parsed.cookies.trim()){
    const ckList=parsed.cookies.split(';').map(s=>s.trim()).filter(Boolean);
    if(ckList.length){
      rows.push({icon:'ck',iconTxt:'🍪',name:'Cookies',count:`${ckList.length} item${ckList.length!==1?'s':''}`});
    }
  }

  /* IndexedDB databases */
  if(parsed.indexedDB){
    const dbs=Object.entries(parsed.indexedDB);
    if(dbs.length){
      dbs.forEach(([dbName,dbData])=>{
        const stores=Object.keys(dbData);
        const totalRows=stores.reduce((acc,s)=>acc+(Array.isArray(dbData[s])?dbData[s].length:0),0);
        rows.push({icon:'idb',iconTxt:'DB',name:dbName,count:`${totalRows} record${totalRows!==1?'s':''}`});
      });
    }
  }

  /* synapse library */
  if(parsed.synapseState?.library?.length){
    rows.push({icon:'ls',iconTxt:'♥',name:'Game Library',count:`${parsed.synapseState.library.length} game${parsed.synapseState.library.length!==1?'s':''}`});
  }

  const date=parsed.exportedAt?new Date(parsed.exportedAt).toLocaleDateString(undefined,{month:'short',day:'numeric',year:'numeric'}):'Unknown date';
  sub.textContent=`Exported ${date}`;

  if(!rows.length){
    list.innerHTML='<div class="il-none">No game data found in this file.</div>';
  } else {
    list.innerHTML=rows.map(r=>`
      <div class="il-row">
        <div class="il-row-icon ${r.icon}">${r.iconTxt}</div>
        <div class="il-row-name">${r.name}</div>
        ${r.count?`<div class="il-row-count">${r.count}</div>`:''}
      </div>`).join('');
  }

  panel.classList.add('show');
  // auto-hide after 12s
  setTimeout(()=>panel.classList.remove('show'),12000);
}

function loadData(){
  const input=document.createElement('input');input.type='file';input.accept='.json';
  input.onchange=e=>{
    const f=e.target.files[0];if(!f)return;
    const r=new FileReader();
    r.onload=async ev=>{
      try{
        const parsed=JSON.parse(ev.target.result);

        const src=parsed.synapseState||parsed;
        Object.assign(state,src);
        if(!state.library)state.library=[];
        persist();

        if(parsed.localStorage){
          for(const [k,v] of Object.entries(parsed.localStorage)){
            try{localStorage.setItem(k,v);}catch(e){}
          }
        }

        if(parsed.cookies){
          for(const pair of parsed.cookies.split(';')){
            const trimmed=pair.trim();if(trimmed)document.cookie=trimmed;
          }
        }

        if(parsed.indexedDB&&Object.keys(parsed.indexedDB).length){
          await restoreIndexedDB(parsed.indexedDB);
        }

        applyTheme(state.theme);
        document.getElementById('panicDisp').textContent=state.panicKey;
        document.getElementById('panicUrlInput').value=state.panicUrl;
        if(state.experience==='full')document.getElementById('navProxy').style.display='';
        state.onboarded=true;persist();
        document.getElementById('onboarding').classList.add('hide');
        renderAll();
        showToast('✓ Data loaded!');
        showImportLog(parsed);
      }catch(err){showToast('Invalid file');}
    };r.readAsText(f);
  };input.click();
}

function goProxy(){let url=document.getElementById('proxyInput').value.trim();if(!url)return;if(!url.startsWith('http'))url='https://'+url;document.getElementById('proxyWrap').innerHTML=`<iframe src="${url}" style="width:100%;height:100%;border:none;" sandbox="allow-scripts allow-same-origin allow-forms"></iframe>`;}
function homeSearchKey(e){
  if(e.key!=='Enter')return;
  const v=document.getElementById('homeSearch').value.trim();if(!v)return;
  const url=v.includes('.')&&!v.includes(' ')?(v.startsWith('http')?v:'https://'+v):`https://www.google.com/search?q=${encodeURIComponent(v)}`;
  if(state.experience==='full'){switchTab('proxy');document.getElementById('proxyInput').value=url;goProxy();}
  else window.open(url,'_blank');
}
